#include<stdio.h>
#include<stdlib.h>

int main(){
    int req[20], n, i, head, seek=0;

    printf("Enter number of disk requests: ");
    scanf("%d",&n);

    printf("Enter request sequence:\n");
    for(i=0;i<n;i++)
        scanf("%d",&req[i]);

    printf("Enter initial head position: ");
    scanf("%d",&head);

    for(i=0;i<n;i++){
        seek += abs(head - req[i]);
        head = req[i];
    }

    printf("\nTotal Seek Time = %d\n", seek);
    return 0;
}
