#include <stdio.h>

int main() {
    int block[10], process[10], allocation[10];
    int nb, np, i, j;

    printf("Enter number of memory blocks: ");
    scanf("%d", &nb);

    printf("Enter sizes of blocks:\n");
    for(i=0;i<nb;i++)
        scanf("%d",&block[i]);

    printf("Enter number of processes: ");
    scanf("%d",&np);

    printf("Enter sizes of processes:\n");
    for(i=0;i<np;i++)
        scanf("%d",&process[i]);

    for(i=0;i<np;i++)
        allocation[i] = -1;

    for(i=0;i<np;i++) {
        for(j=0;j<nb;j++) {
            if(block[j] >= process[i]) {
                allocation[i] = j;
                block[j] -= process[i];
                break;
            }
        }
    }

    printf("\nProcess No\tProcess Size\tBlock No\n");
    for(i=0;i<np;i++) {
        if(allocation[i] != -1)
            printf("%d\t\t%d\t\t%d\n", i+1, process[i], allocation[i]+1);
        else
            printf("%d\t\t%d\t\tNot Allocated\n", i+1, process[i]);
    }

    return 0;
}
