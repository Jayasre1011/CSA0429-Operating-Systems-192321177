#include<stdio.h>

int main(){
    int indexBlock[20], n, i;

    printf("Enter number of blocks required for file: ");
    scanf("%d",&n);

    printf("Enter block numbers:\n");
    for(i=0;i<n;i++)
        scanf("%d",&indexBlock[i]);

    printf("\nIndex Block Allocation Table\n");
    printf("File Block No\tDisk Block No\n");

    for(i=0;i<n;i++)
        printf("%d\t\t%d\n", i+1, indexBlock[i]);

    return 0;
}
