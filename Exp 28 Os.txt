#include <stdio.h>
#include <string.h>

int main(){
    char line[100], word[20];

    printf("Enter a line of text:\n");
    fgets(line,100,stdin);

    printf("Enter word to search: ");
    scanf("%s",word);

    if(strstr(line,word))
        printf("Word Found\n");
    else
        printf("Word Not Found\n");

    return 0;
}
