#include<stdio.h>

int main(){
    int block[20], n, i;

    printf("Enter number of blocks required: ");
    scanf("%d",&n);

    printf("Enter block numbers in sequence:\n");
    for(i=0;i<n;i++)
        scanf("%d",&block[i]);

    printf("\nLinked Allocation:\n");
    for(i=0;i<n-1;i++)
        printf("%d -> ", block[i]);
    printf("%d\n", block[n-1]);

    return 0;
}
