#include <stdio.h>
#include <string.h>

int main() {
    char files[20][20];
    int n,i;

    printf("Enter number of files: ");
    scanf("%d",&n);

    for(i=0;i<n;i++) {
        printf("Enter file name %d: ",i+1);
        scanf("%s",files[i]);
    }

    printf("\nFiles in Directory:\n");
    for(i=0;i<n;i++)
        printf("%s\n",files[i]);

    return 0;
}
